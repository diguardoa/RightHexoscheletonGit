23// Conversions between SL_Bus_proxy_sensor_msgs_JointState and sensor_msgs::JointState



24



25void convertFromBus(sensor_msgs::JointState* msgPtr, SL_Bus_proxy_sensor_msgs_JointState const* busPtr)



26{



27  const std::string rosMessageType("sensor_msgs/JointState");



28



29  convertFromBusVariablePrimitiveArray(msgPtr->effort, busPtr->Effort, busPtr->Effort_SL_Info);



30  convertFromBus(&msgPtr->header, &busPtr->Header);



31  convertFromBusVariableStringArray(msgPtr->name, busPtr->Name, busPtr->Name_SL_Info);



32  convertFromBusVariablePrimitiveArray(msgPtr->position, busPtr->Position, busPtr->Position_SL_Info);



33  convertFromBusVariablePrimitiveArray(msgPtr->velocity, busPtr->Velocity, busPtr->Velocity_SL_Info);



34}
